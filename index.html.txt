function safeJSON(text) {
  try {
    const c = text.replace(/^```json\s*/i,"").replace(/^```\s*/i,"").replace(/```\s*$/i,"").trim();
    return JSON.parse(c);
  } catch {
    const a = text.match(/\[[\s\S]*\]/);
    if(a) try{return JSON.parse(a[0]);}catch{}
    const o = text.match(/\{[\s\S]*\}/);
    if(o) try{return JSON.parse(o[0]);}catch{}
    return null;
  }
}

async function askClaude(apiKey, system, user) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method:"POST",
    headers:{
      "content-type":"application/json",
      "x-api-key": apiKey,
      "anthropic-version":"2023-06-01",
      "anthropic-dangerous-direct-browser-access":"true"
    },
    body: JSON.stringify({
      model:"claude-sonnet-4-20250514",
      max_tokens:8000,
      system,
      messages:[{role:"user",content:user}]
    })
  });
  const d = await res.json();
  if(d.error) throw new Error(d.error.message);
  return d.content.map(b=>b.text||"").join("").trim();
}

async function generateTTS(apiKey, input, onProgress) {
  const date = new Date().toLocaleDateString("az-AZ");

      prompt:`Elicitation:\n${input}\n\nJSON doldur:\n{"stakeholders":[{"role":"...","name":"TBD","responsibility":"..."}],"users":[{"category":"...","description":"...","access":"..."}]}`
    },
    {
      msg:"📖 Terminlər, abbreviaturalar, proses...", key:"block4",
      prompt:`Elicitation:\n${input}\n\nJSON doldur:\n{"terms":[{"term":"...","definition":"..."}],"acronyms":[{"acronym":"...","description":"..."}],"process":"..."}`
    },
    {
      msg:"🔗 İnteqrasiyalar və asılılıqlar...", key:"block5",
      prompt:`Elicitation:\n${input}\n\nJSON doldur:\n{"external":[{"system":"...","type":"REST API","description":"..."}],"internal":[{"system":"...","type":"...","description":"..."}],"dependencies":[{"item":"...","description":"..."}]}`
    },
    {
      msg:"⚙️ Servis detalları...", key:"block6",
      prompt:`Elicitation:\n${input}\n\nJSON doldur (bodyExample sadə string olsun):\n{"method":"POST","url":"/api/v1/...","description":"...","bodyExample":"field1: value1","successResponse":"status: success","errorResponse":"code: 400, message: ...","errorCodes":[{"code":"400","description":"..."}]}`
    },
    {
      msg:"✅ Funksional və qeyri-funksional tələblər...", key:"block7",
      prompt:`Elicitation:\n${input}\n\nJSON doldur:\n{"functional":[{"id":"FR-001","requirement":"...","priority":"High"}],"nonfunctional":[{"id":"NFR-001","requirement":"...","category":"Performance"}]}`
    },
    {
      msg:"⚠️ Risklər və qəbul meyarları...", key:"block8",
      prompt:`Elicitation:\n${input}\n\nJSON doldur:\n{"technical":[{"risk":"...","impact":"...","mitigation":"..."}],"business":[{"risk":"...","impact":"...","mitigation":"..."}],"acceptance":[{"id":"AC-001","criteria":"...","testCase":"..."}]}`
    }
  ];

  for(let i=0;i<calls.length;i++){
    onProgress(calls[i].msg, i+1);
    const raw = await askClaude(apiKey, SYS, calls[i].prompt);
    const p = safeJSON(raw);
    if(!p) continue;
    if(i===0) { result.general={documentTitle:p.documentTitle,documentVersion:p.documentVersion,documentDate:date,itBusinessAnalysts:"IT BA",jiraLink:"—"}; result.summary=p.summary; }
    else if(i===1) { result.asis=p.asis; result.businessneed=p.businessneed; result.tobe=p.tobe; }
    else if(i===2) { result.stakeholders=p.stakeholders||[]; result.users=p.users||[]; }
    else if(i===3) { result.terms=p.terms||[]; result.acronyms=p.acronyms||[]; result.process=p.process; }
    else if(i===4) { result.integrations={external:p.external||[],internal:p.internal||[],dependencies:p.dependencies||[]}; }
    else if(i===5) { result.service=p; }
    else if(i===6) { result.functional=p.functional||[]; result.nonfunctional=p.nonfunctional||[]; }
    else if(i===7) { result.risks={technical:p.technical||[],business:p.business||[]}; result.acceptance=p.acceptance||[]; }
  }
  return result;
}

function Tbl({heads,rows}){
  return(
    <div style={{overflowX:"auto",marginBottom:16,borderRadius:8,border:"1px solid #e2e8f0",overflow:"hidden"}}>
      <table style={{width:"100%",borderCollapse:"collapse",fontSize:13}}>

            {heads.map(h=><th key={h} style={{padding:"10px 14px",color:"white",textAlign:"left",fontWeight:600}}>{h}</th>)}
          </tr>
        </thead>
        <tbody>
          {(!rows||rows.length===0)
            ?<tr><td colSpan={heads.length} style={{padding:"12px 14px",color:"#94a3b8",fontStyle:"italic"}}>Məlumat yoxdur</td></tr>
            :(rows||[]).map((row,i)=>(

              </tr>
            ))
          }
        </tbody>
      </table>
    </div>
  );
}
